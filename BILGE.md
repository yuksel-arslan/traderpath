```typescript

const KNOWN_PATTERNS = {
  // Null/Undefined Errors
  'NULL_ACCESS': {
    pattern: /Cannot read propert(y|ies) .* of (undefined|null)/,
    category: 'NULL_ACCESS',
    severity: 'HIGH',
    commonCauses: [
      'Interface field name mismatch',
      'Missing null check',
      'Async data not loaded yet',
      'Optional chaining needed'
    ],
    suggestedFix: 'Check field names match interface. Add optional chaining (?.) or null check.',
    exampleFix: 'object.field?.method() || defaultValue'
  },

  // Connection Errors
  'DB_CONNECTION': {
    pattern: /ECONNREFUSED|connection refused|unable to connect/i,
    category: 'CONNECTION',
    severity: 'CRITICAL',
    commonCauses: [
      'Database server down',
      'Connection string incorrect',
      'Firewall blocking',
      'Connection pool exhausted'
    ],
    suggestedFix: 'Check database status, connection string, and firewall rules.',
    autoAction: 'CHECK_DB_STATUS'
  },

  // Rate Limiting
  'RATE_LIMIT': {
    pattern: /429|too many requests|rate limit/i,
    category: 'RATE_LIMIT',
    severity: 'MEDIUM',

      'API quota exceeded',
      'Missing request throttling',
      'Burst traffic'
    ],
    suggestedFix: 'Implement exponential backoff. Check API quota.',
    autoAction: 'ENABLE_BACKOFF'
  },

  // Type Errors

    pattern: /TypeError|is not a function|is not iterable/,

    commonCauses: [
      'Wrong data type passed',
      'Interface mismatch',
      'Missing type conversion'
    ],
    suggestedFix: 'Check type definitions and add proper type guards.'
  },

  // Memory Issues
  'MEMORY': {
    pattern: /heap out of memory|ENOMEM|memory limit/i,

      'Memory leak',
```
